30 // The logger can act straight away (when Logger::print is called), but for performance reasons it is better to
31 // buffer the log messages and dump them en masse periodically, preferably in a background thread.
32 // Unfortuantely there are some hairy object lifetime difficulties involved in keeping our *own* thread, since the
33 // logger is a static singleton, which will be destroyed after main() has exited, which, with VS2013 at least,
34 // will cause a deadlock our thread hasn't already joined() -
35 // see https://connect.microsoft.com/VisualStudio/feedback/details/747145 for example.
36 // See comments below in log_spooler() for details.

88  Logger() : line_number(0), indentation(0)
89  {
90  this->has_backend = false;
91  this->is_spooling.clear();
92  this->logger_backend = new std::thread{ Logger::log_spooler, this };
93  }
94  ~Logger()
95  {
96  // Switch off logging now, to avoid complications:
97  this->severity_level = LOG_OFF;
98  // Let our spooling thread know that we want it to stop:
99  this->is_spooling.clear();
100  // Wait for it to finish (we can't use join() due to
101  // the exit lock issue.)
102  // In some scenarios, ExitProcess has already been called by this point,
103  // and our thread will have been terminated - so we don't want to wait
104  // indefinitely.
105  // It would be nicer to use std::this_thread::sleep_for(...), but it's not safe
106  // to call from within dllmain.
107  auto start = std::chrono::system_clock::now();
108  while (this->has_backend && (std::chrono::system_clock::now() - start).count() < 2.0);
109  // Clear whatever is left in our buffer.
110  // DON'T acquire the write_guard lock at this point - by this point
111  // there should be no danger of anyone else accessing our buffer,
112  // and if we got here because the process is exiting, it might not be
113  // safe to use the mutex.
114  clear_backlog();
115  // Detach the thread:
116  this->logger_backend->detach();
117  // And close our file, if we have one:
118  if (this->writer.is_open())
119  this->writer.close();
120  }

187  static void log_spooler(Logger* logger)
188  {
189  logger->has_backend = true;
190  logger->is_spooling.test_and_set();
191  std::unique_lock<std::timed_mutex> writing_lock{ logger->write_guard, std::defer_lock };
192  // Loop until the logger clears the is_spooling flag - this happens when the logger is destructed.
193  do
194  {
195  std::this_thread::sleep_for(std::chrono::milliseconds{ 100 });
196  if (logger->log_buffer.size())
197  {
198  writing_lock.lock();
199  logger->clear_backlog();
200  writing_lock.unlock();
201  }
202  } while (logger->is_spooling.test_and_set());
203  // The logger will be waiting for us to finish spooling - let it know it can continue:
204  logger->has_backend = false;
205  // ON WINDOWS, VS2013:
206  // If we let this function end now, we will get a nasty race condition with the CRT.
207  // When the thread finishes running its main method (this function), it will enter
208  // _Cnd_do_broadcast_at_thread_exit(), and attempt to signal the news of the thread's demise
209  // to anyone waiting for it. Doing this involves creating an "at_thread_exit_mutex", and adding
210  // a destruction method for it ("destroy_at_thread_exit_mutex") to the DLL's atexit/_onexit list.
211  // (This is a list of functions which will get called when the DLL is signalled to exit by ExitProcess.)
212 
213  // BUT:
214 
215  // Because logger is a static singleton, its destruction takes place AFTER main() exits, at
216  // which point the CRT has imposed an EXIT LOCK. All exit code needs to own the exit lock before it can
217  // execute. At the point when ~Logger() is called, the main thread owns this exit lock, and it won't
218  // relinquish it until right before it calls ExitProcess, after going through its own atexit/_onexit list.
219  // The worker thread needs to aquire this lock before it can create the at_thread_exit_mutex, so it
220  // is forced to wait while everything else is destructed and cleaned before it can carry on.
221 
222  // THe end result is that the worker thread has just enough time to create and lock the mutex before
223  // the main thread calls ExitProcess, which calls the DLL cleanup code, which deletes the mutex,
224  // resulting in an abort() with a "mutex destroyed while busy" error.
225 
226  // To get around this, we simply go to sleep for a bit, to give the main thread a chance to exit
227  // before we even have a chance to leave this function.
228  std::this_thread::sleep_for(std::chrono::milliseconds(100000));
229  // It'll all be over before we even wake up.
230  }

279  template<Logger::LoggingSeverityLevel level, typename...Args>void Logger::print(Args&&...args)
280  {
281  if (level > severity_level)
282  return;
283  std::stringstream message_stream;
284  auto now = boost::posix_time::microsec_clock::universal_time();
285  message_stream << now << " P "; // 'P' for 'Platform' - useful if combining logs with Mod-side.
286  switch (level)
287  {
288  case LoggingSeverityLevel::LOG_ALL:
289  case LoggingSeverityLevel::LOG_ERRORS:
290  message_stream << "ERROR ";
291  break;
292  case LoggingSeverityLevel::LOG_WARNINGS:
293  message_stream << "WARNING ";
294  break;
295  case LoggingSeverityLevel::LOG_INFO:
296  message_stream << "INFO ";
297  break;
298  case LoggingSeverityLevel::LOG_FINE:
299  message_stream << "FINE ";
300  break;
301  case LoggingSeverityLevel::LOG_TRACE:
302  case LoggingSeverityLevel::LOG_OFF:
303  message_stream << "TRACE ";
304  break;
305  }
306  for (int i = 0; i < this->indentation; i++)
307  message_stream << " ";
308 
309  print_impl(std::forward<std::stringstream>(message_stream), std::move(args)...);
310  this->line_number++;
311  }

330  class LoggerLifetimeTracker

332  public:

347  private:
348  void addref()
349  {
350  int prev_val = object_count.fetch_add(1);
351  LOGFINE(LT("Constructing "), this->name, LT(" (object count now "), prev_val + 1, LT(")"));
352  }
